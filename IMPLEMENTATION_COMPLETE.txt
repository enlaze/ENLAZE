  - Network errors: 2 second delay
  - Timeouts: 3 second delay
  - Rate limits (429): 5 second delay
  - Server errors (5xx): 2 second delay
  - Auth/permission: No retry (user action needed)
  - Unknown errors: No retry

HOW TO USE
==========

In Pages/Components:
  import { logError, formatErrorForUI } from "@/lib/error-handler";
  import ErrorAlert from "@/components/ErrorAlert";
  import ErrorBoundary from "@/components/ErrorBoundary";

  // Try-catch with logging
  try {
    await fetchData();
  } catch (err) {
    logError(err, { component: "MyPage", action: "fetch" });
    setError(err);
  }

  // Display error
  {error && <ErrorAlert title="..." message="..." variant="error" />}

  // Wrap risky section
  <ErrorBoundary name="my-section">
    <RiskyComponent />
  </ErrorBoundary>

In Forms:
  try {
    await submitForm();
  } catch (err) {
    logError(err, { component: "MyForm", action: "submit" });
    setError(err);
  }

  {error && (
    <ErrorAlert
      title="Submission Failed"
      message={error.message}
      action={{ label: "Retry", onClick: handleSubmit }}
    />
  )}

ERROR HANDLING FLOWS
====================

Global Error:
  Uncaught Error → app/error.tsx → Professional error page

Dashboard Error:
  Error in dashboard → app/dashboard/error.tsx → Context-aware error page

Component Error:
  Error in section → ErrorBoundary → ErrorAlert + Reset button

Data Loading Error:
  try { fetch } catch { logError, setState, render ErrorAlert }

Form Submission Error:
  try { submit } catch { logError, setState, show inline ErrorAlert }
